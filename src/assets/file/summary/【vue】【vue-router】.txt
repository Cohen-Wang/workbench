【编程式传参】
<button @click="">跳转</button>

methods: {
    goWhere() {
        // 不传参，写法一
        this.router.push('/about')

        // 不传参，写法二
        this.router.push({name: 'about'})

        // 传参1
        this.router.push({name: 'about', params: {name: 'cohen'}})

        // 传参2
        this.router.push({path: '/about', query: {name: 'cohen'}})
    }
}
